leetcode

二分模板

>先写一个check函数
>
>判定在check的情况下（true和false的情况下），如何更新区间。
>
>在check(m)==true的分支下是:
>
>​	l=mid的情况，中间点的更新方式是m=(l+r+1)/2       r = mid - 1，此时寻找的是最后一个满足条件的额值
>
>​	r=mid的情况，中间点的更新方式是m=(l+r)/2      l = mid +1 ，此时寻找的是第一个满足条件的值

```cpp
//寻找第一个满足条件的值
// 区间[l, r]被划分成[l, mid]和[mid + 1, r]时使用：
int bsearch_1(int l, int r)
{
    while (l < r)
    {
        int mid = l + r >> 1;
        if (check(mid)) r = mid;    // check()判断mid是否满足性质
        else l = mid + 1;
    }
    return l;
}
//寻找最后一个满足条件的值
// 区间[l, r]被划分成[l, mid - 1]和[mid, r]时使用：
int bsearch_2(int l, int r)
{
    while (l < r)
    {
        int mid = l + r + 1 >> 1;
        if (check(mid)) l = mid;
        else r = mid - 1;
    }
    return l;
}
```
