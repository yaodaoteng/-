struct Node{
    int l;
    int r;
}node[N];

int cmp(struct Node a, struct Node b){
    if(a.l == b.l) return a.r < b.r;
    return a.l < b.l;
}

        if(node[i].l > right){ //如果当前区间的左边界大于之前一个区间的右边界，则区间数加一，同时更新区间左右边界
            res++;
            left = node[i].l;
            right = node[i].r;
